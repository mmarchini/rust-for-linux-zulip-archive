pub fn version_number_entries(&self) -> &[u16] {
    let ptr = self.as_ptr();
    let len = unsafe { (*ptr).version_number_entry_count };
    let arr_ptr = unsafe { addr_of!((*ptr).version_number_entries) };
    unsafe { core::slice::from_raw_parts(arr_ptr.cast::<u16>(), len) }
}
