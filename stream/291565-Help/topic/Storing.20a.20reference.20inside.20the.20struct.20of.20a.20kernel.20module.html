Storing an immutable references to bindings::module without wrapping it in Opaque is problematic because it comes with an assumption that the module is immutable. The compiler may optimize under the assumption that reading any field of the module twice results in the same value for both reads, which is probably an incorrect assumption.

use kernel::bindings;
use kernel::bindings::*;
use kernel::str::*;
use kernel::*;

pub fn safe_deref<'a,T>(ptr: *const T) -> Option<&'a T> {
    if ptr.is_null() {
        None
    } else {
        unsafe { Some(&*ptr) }
    }
}

pub fn safe_deref_mut<'a, T>(ptr: *mut T) -> Option<&'a mut T> {
    if ptr.is_null() {
        None
    } else {
        unsafe { Some(&mut *ptr) }
    }
}

pub fn find_module(module_name: &CString) -> Option<&module>
{
    unsafe
    {
        return safe_deref(bindings::find_module(module_name.as_char_ptr()));
    };
}

pub fn get_task_cred(task: &mut task_struct) -> Option<&cred>
{
    unsafe
    {
        return safe_deref(bindings::get_task_cred(task as *mut task_struct));
    }
}
