// SPDX-License-Identifier: GPL-2.0

//! Unsafe Precondition assert.

/// Assert that an unsafe function's preconditions hold.
///
/// This macro should only ever be used inside an `unsafe fn` to enforce
/// expectations on the caller.
///
/// # Examples
///
/// ```ignore
/// // SAFETY: `ptr` must not be null.
/// pub unsafe fn do_something(ptr: *const u8) {
///     unsafe_precondition_assert!(!ptr.is_null(), "ptr must not be null");
/// }
/// ```
#[macro_export]
macro_rules! unsafe_precondition_assert {
    ($cond:expr $(,)?) => {
        ::core::debug_assert!(
            $cond,
            "unsafe precondition assertion failed: {}",
            stringify!($cond)
        );
    };
    ($cond:expr, $fmt:expr $(, $arg:expr)* $(,)?) => {
        ::core::debug_assert!(
            $cond,
            concat!("unsafe precondition assertion failed: ", $fmt)
            $(, $arg)*
        );
    };
}

and am currently trying to find where i can use this assertion and believe that the SAFETY: comments would be a good place to start using this macro, asserting that the safety properties are true.

Though a lot of where I see these being used, didn't have assertions to begin with and I am not sure if would be appropriate?

For example, in device.rs there's a safety comment to ensure the ptr is not null. Would this be a good place to start using the macro, e.g.

    pub unsafe fn get_device(ptr: *mut bindings::device) -> ARef<Self> {

        unsafe { Self::as_ref(ptr) }.into()
    }
