use core::panic::PanicInfo;

include!("bindings.rs");

#[panic_handler]
fn handle_panic(_info: &PanicInfo) -> ! {
    loop {}
}

extern "C" {
    static mut ovpn_link_ops: rtnl_link_ops;
    fn ovpn_tcp_init() -> i32;
    fn ovpn_netlink_register() -> i32;
    fn ovpn_netlink_unregister();
}

#[no_mangle]
pub extern "C" fn ovpn_init() -> i32 {
    unsafe {
        let err = ovpn_tcp_init();
        if err != 0 {
            return err;
        }

        let err = rtnl_link_register(&mut ovpn_link_ops);
        if err != 0 {
            return err;
        }

        let err = ovpn_netlink_register();
        if err != 0 {
            return err;
        }
    }

    0
}

#[no_mangle]
pub extern "C" fn ovpn_cleanup() {
    unsafe {
        rtnl_link_unregister(&mut ovpn_link_ops);
        ovpn_netlink_unregister();
        rcu_barrier(); // Ensure all RCU callbacks are completed
    }
}
