Dirk Behme (Feb 01 2024 at 12:35):

Maybe anything like this (see Playground) ?

pub struct TheStruct {
    pub foo: [::std::os::raw::c_char; 100usize],
    // other fields...
}

pub fn main() {
    dbg!( get_size_of_return_type(|s: TheStruct| s.foo) );
}

fn get_size_of_return_type<F, T, U>(_f: F) -> usize
where
    F: FnOnce(T) -> U
{
    let size = std::mem::size_of::<U>(); // For kernel code replace std by core
    if size > 90 {
        compile_error!("Size too large");
    }
    size
}

Inspired by the given stackoverflow thread.
