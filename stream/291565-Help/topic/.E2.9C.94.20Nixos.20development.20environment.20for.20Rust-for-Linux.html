Marcel Wiesenhöfer (Jul 03 2024 at 22:42):

I use this flake.nix:

{
  description = "A nix flake for compiling the kernel with rust";
  inputs = {
    nixpkgs_old.url =
      "github:NixOS/nixpkgs/2d5c19598ad493b31d522ba89d2e683c5964b69a";
    nixpkgs.url = "github:NixOS/nixpkgs/nixos-unstable";
    rust-overlay.url = "github:oxalica/rust-overlay";
  };

  outputs = { self, nixpkgs, nixpkgs_old, rust-overlay }:
    let
      overlays = [ (import rust-overlay) ];
      system = "x86_64-linux";
      pkgs = import nixpkgs { inherit system overlays; };
      pkgs_old = import nixpkgs_old { inherit system overlays; };
      rust = pkgs.rust-bin.stable."1.78.0".default.override {
        extensions = [ "rust-src" "rust-analyzer" ];
      };
      rust_platform = pkgs.makeRustPlatform {
        cargo = rust;
        rustc = rust;
      };
    in {
      devShells.${system}.default =
        pkgs.mkShell.override { stdenv = pkgs.clangStdenv; } {
          buildInputs = [
            rust
            (pkgs_old.rust-bindgen.override {
              rust-bindgen-unwrapped =
                pkgs_old.rust-bindgen-unwrapped.override {
                  rustPlatform = rust_platform;
                  clang = pkgs.clang;
                };
            })
            pkgs.util-linux
            (pkgs.llvmPackages.bintools.override {
              sharedLibraryLoader = null;
            })
            pkgs.elfutils
            pkgs.flex
            pkgs.bison
            pkgs.bc
            pkgs.perl
            pkgs.openssl
            pkgs.ncurses
            pkgs.bear
          ];
        };

    };
}

Depending on the kernel sources you may have to edit the rust version.
The specific nixpkgs commit is for bindgen 0.65.1.
Some packges may be missing or not needed depending on your uses

you would use this with nix develop

Andreas Hindborg (Jul 04 2024 at 07:39):

I do something like this:

    rustc-version = pkgs.lib.strings.removeSuffix "\n" (builtins.readFile (
      pkgs.runCommand "rustc-version" {} " cd ${cfg.src}; scripts/min-tool-version.sh rustc > $out"));
    kernelRust = rust-pkgs.rust-bin.stable."${rustc-version}".complete;
    llvm-version = pkgs.lib.strings.removeSuffix "\n" (builtins.readFile (
      pkgs.runCommand "llvm-version" {} ''
        ${kernelRust}/bin/rustc --version --verbose | grep 'LLVM version' | sed -r 's/.* ([0-9]+)\.[0-9]+\.[0-9]/\1/' > $out
      ''));
    kernelLlvm = pkgs.buildPackages."llvmPackages_${llvm-version}";

    rawKernel = pkgs.callPackage ./kernel.nix {
      inherit
        pkgs
        lib
        kernelLlvm
        kernelRust
        kernelBindgen;
      stdenv = pkgs.stdenv;
      kernelSrc = cfg.src;
      configFragments = cfg.fragments;
    };

    kernel = lib.recurseIntoAttrs (pkgs.linuxKernel.packagesFor rawKernel);

Then you can use it like a regular kernel and build a regular NixOS system closure with it.
