pub struct DmaFence<T> {

    data: T,

        let fence = pin_init!(Self {
            inner <- Opaque::ffi_init(|slot: *mut bindings::dma_fence| {

                // reference of the fence context.

            }),
            data <- data,
            signalling: false,
            signalling_cookie: false,
            fctx: fctx,
        });
