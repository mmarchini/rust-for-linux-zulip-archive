Timothy G. (Oct 18 2024 at 10:32):

If one initialises a rust enum with the same values as the C enum, there is a collision of multiple enum values because
HRTIMER_MODE_ABS|<anything> == <anything>
And so you cannot compile because the enum has, for instance,
TimerMode::Pinned == TimerMode::AbsolutePinned

Changing the values so that MODE_ABS combos have different values than without MODE_ABS allows the identical initialisation of the rust enum + then you can easily get the C value from the enum since they are the same.

the Rust code that does this:

#[derive(Debug)]
pub struct TimerMode(u8);
impl TimerMode {
    pub fn builder() -> Builder<Missing<AbsoluteRelative>, Missing<Pinned>, Missing<HardSoft>> {
        Builder {
            flags: Default::default(),
            t: std::marker::PhantomData,
        }
    }
}

#[derive(Debug)]
#[repr(C)]
/// If I understand correcly, when it is Repr(C) is is safe to transmute the builder to change only
/// unit generics held as PhandomData. If it is not Repr(C) there are no solid garanties that the
/// memory layout does not change depending on the unit generic.
/// A safe alternative is to make new different instance with the right generics and copy/move the
/// flags into it. If creating TimerModes is not a hot path then maybe the safer way is best.
pub struct Builder<S1, S2, S3> {
    flags: [Option<u8>; 3],
    t: std::marker::PhantomData<(S1, S2, S3)>,
}

#[derive(Debug)]
pub struct Missing<T> {
    t: std::marker::PhantomData<T>,
}

#[derive(Debug)]
pub struct Valid<T> {
    t: std::marker::PhantomData<T>,
}

#[derive(Debug)]
pub struct AbsoluteRelative;
#[derive(Debug)]
pub struct Pinned;
#[derive(Debug)]
pub struct HardSoft;

impl<S2, S3> Builder<Valid<AbsoluteRelative>, S2, S3> {
    pub fn set_absolute(&mut self) {
        self.flags[0] = Some(HRTIMER_MODE_ABS);
    }
    pub fn set_relative(&mut self) {
        self.flags[0] = Some(HRTIMER_MODE_REL);
    }
}

impl<S1, S3> Builder<S1, Valid<Pinned>, S3> {
    pub fn set_pinned(&mut self) {
        self.flags[1] = Some(HRTIMER_MODE_PINNED);
    }
    pub fn set_unpinned(&mut self) {
        self.flags[1] = Some(0);
    }
}

impl<S1, S2> Builder<S1, S2, Valid<HardSoft>> {
    pub fn set_hard(&mut self) {
        self.flags[2] = Some(HRTIMER_MODE_HARD);
    }
    pub fn set_soft(&mut self) {
        self.flags[2] = Some(HRTIMER_MODE_SOFT);
    }
}

impl Builder<Valid<AbsoluteRelative>, Valid<Pinned>, Valid<HardSoft>> {
    pub fn build(self) -> TimerMode {
        TimerMode(self.flags.iter().flatten().sum::<u8>())
    }
}

impl<S2, S3> Builder<Missing<AbsoluteRelative>, S2, S3> {
    pub fn with_absolute(self) -> Builder<Valid<AbsoluteRelative>, S2, S3> {
        let mut b: Builder<Valid<AbsoluteRelative>, S2, S3> = unsafe { std::mem::transmute(self) };
        b.set_absolute();
        b
    }

    pub fn with_relative(self) -> Builder<Valid<AbsoluteRelative>, S2, S3> {
        let mut b: Builder<Valid<AbsoluteRelative>, S2, S3> = unsafe { std::mem::transmute(self) };
        b.set_relative();
        b
    }
}

impl<S1, S3> Builder<S1, Missing<Pinned>, S3> {
    pub fn with_pinned(self) -> Builder<S1, Valid<Pinned>, S3> {
        let mut b: Builder<S1, Valid<Pinned>, S3> = unsafe { std::mem::transmute(self) };
        b.set_pinned();
        b
    }

    pub fn with_unpinned(self) -> Builder<S1, Valid<Pinned>, S3> {
        let mut b: Builder<S1, Valid<Pinned>, S3> = unsafe { std::mem::transmute(self) };
        b.set_unpinned();
        b
    }
}
impl<S1, S2> Builder<S1, S2, Missing<HardSoft>> {
    pub fn with_hard(self) -> Builder<S1, S2, Valid<HardSoft>> {
        let mut b: Builder<S1, S2, Valid<HardSoft>> = unsafe { std::mem::transmute(self) };
        b.set_hard();
        b
    }

    pub fn with_soft(self) -> Builder<S1, S2, Valid<HardSoft>> {
        let mut b: Builder<S1, S2, Valid<HardSoft>> = unsafe { std::mem::transmute(self) };
        b.set_soft();
        b
    }
}

fn main() {
    let mut mode_builder = TimerMode::builder().with_absolute().with_hard();
    mode_builder.set_soft();
    let mode = mode_builder.with_unpinned().build();
    println!("{:?}", mode);
}

No, I just imitated the modules proc_macro at first for the initial typestate builder schema bitflag proc_macro.

Then the OpenOptions-like versionbitflag_options proc_macro is simpler since I got rid of typestate building (because it has default values it is always in a valid state + calling relative(true).relative(false) is equal to calling relative(false)).

Finally, since I got rid of the typestate builder, i did not need to use proc_macro for the compicated variations of <(Valid|Invalid)+> states of the builder. It ends up fitting in a much simpler and cleaner in my opinion:

macro_rules! bitflag_options_rules {{

    name:$name:ident,
    type:$t:ty,
    options:[
        $({name:$fn_name:ident, true: $true_bitval:expr, false: $false_bitval:expr}),+
]} => {

        #[derive(Debug, PartialEq)]
        pub struct $name($t);

        impl BitFlag for $name {
        type Bits = $t;

        fn bits(&self) -> Self::Bits {
            self.0
        }
    }

    impl $name {
    $(pub fn $fn_name(mut self, $fn_name:bool)->Self{
            if $fn_name{
                self.0 = self.0 & !$false_bitval | $true_bitval;
            }else{
                self.0 = self.0 & !$true_bitval | $false_bitval;
            }
            self
        })+


    }

    impl Default for $name{
        fn default() -> Self {
            Self(0)$(.$fn_name(false))+
        }
    }

    impl TryFrom<<$name as BitFlag>::Bits> for $name {
        type Error = <$name as BitFlag>::Bits;

        fn try_from(value: <$name as BitFlag>::Bits) -> Result<Self, Self::Error> {
            let mut to_process = value.clone();

            let mut $(
            matched = false;

            for flag in [$true_bitval, $false_bitval] {
                if (flag & to_process) == flag {
                    matched = true;
                    to_process -= flag;
                    if flag > 0 {
                        break;
                    }
                }
            }
            if !matched {
                return Err(value);
            })+


            if to_process == 0 {
                return Ok(Self(value));
            }
            return Err(value)
        }
    }

    };
}
