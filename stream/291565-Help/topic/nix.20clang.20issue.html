    llvm = pkgsDefault.llvmPackages_18;
    kernelMake = pkgsDefault.writeShellScriptBin "kernel-make" ''
      export HOST_CFLAGS="-I${pkgsDefault.elfutils}/include"
      export LIBCLANG_PATH="${pkgsDefault.llvmPackages_18.libclang.lib}/lib"
      export KBUILD_BUILD_TIMESTAMP="0"
      export KBUILD_BUILD_USER="user"
      export KBUILD_BUILD_HOST="host"

      ${pkgsDefault.gnumake}/bin/make \
        CC="${llvm.clang.cc}/bin/clang" \
          HOSTCC="${llvm.clang}/bin/clang" \
        HOSTCXX="${llvm.clang}/bin/clang++" \
        LD="${llvm.tools.lld}/bin/ld.lld" \
        AR="${llvm.bintools}/bin/ar" \
        NM="${llvm.bintools}/bin/nm" \
        STRIP="${pkgsDefault.stdenv.cc.bintools}/bin/strip" \
          OBJCOPY="${llvm.bintools}/bin/objcopy" \
        OBJDUMP="${llvm.bintools}/bin/objdump" \
        READELF="${llvm.bintools}/bin/readelf" \
        HOSTAR="${llvm.bintools}/bin/ar" \
        HOSTLD="${llvm.tools.lld}/bin/ld.lld" \
        LLVM_AS="${llvm.libllvm}/bin/llvm-as" \

I setup environment by wrapping make like so:

          kernelMake = pkgs.writeShellScriptBin "kernel-make" ''
            #export HOST_CFLAGS="-I${pkgs.elfutils}/include"
            export LIBCLANG_PATH="${llvm.libclang.lib}/lib"
            export KBUILD_BUILD_TIMESTAMP="0"
            export KBUILD_BUILD_USER="user"
            export KBUILD_BUILD_HOST="host"

            ${pkgs.gnumake}/bin/make \
              CC="${cc}/bin/clang" \
              HOSTCC="${llvm.clang}/bin/clang" \
              HOSTCXX="${llvm.clang}/bin/clang++" \
              LD="${llvm.tools.lld}/bin/ld.lld" \
              AR="${llvm.bintools-unwrapped}/bin/llvm-ar" \
              NM="${llvm.bintools-unwrapped}/bin/llvm-nm" \
              STRIP="${llvm.bintools-unwrapped}/bin/llvm-strip" \
              OBJCOPY="${llvm.bintools-unwrapped}/bin/llvm-objcopy" \
              OBJDUMP="${llvm.bintools-unwrapped}/bin/llvm-objdump" \
              READELF="${llvm.bintools-unwrapped}/bin/llvm-readelf" \
              HOSTAR="${llvm.bintools-unwrapped}/bin/llvm-ar" \
              HOSTLD="${llvm.tools.lld}/bin/ld.lld" \
              BINDGEN="${bindgen}/bin/bindgen" \
              LLVM_AS="${llvm.libllvm}/bin/llvm-as" \
              LLVM_LINK="${llvm.libllvm}/bin/llvm-link" \
              ARCH="${kernelArch}" \
              RUSTFMT="${rustfmt}/bin/rustfmt" \
              $@
          '';
