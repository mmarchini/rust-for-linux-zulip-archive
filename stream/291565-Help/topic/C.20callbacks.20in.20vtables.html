Boqun Feng (Jul 31 2023 at 16:35):

How about a DefaultOperationSelection trait?

pub trait DefaultOperationSelection {
   const READ: Option<
        unsafe extern "C" fn(
            file: *mut bindings::file,
            buf: *mut core::ffi::c_char,
            len: core::ffi::c_size_t,
            offset: *mut bindings::loff_t
         ) -> core::ffi::c_ssize_t> = None;
   ...
}

and 

impl<A: OpenAdapter<T::OpenData>, T: Operations + DefaultOperationSelection> OperationsVtable<A, T> {
    const VTABLE: bindings::file_operations = bindings::file_operations {
        open: Some(Self::open_callback),
        release: Some(Self::release_callback),
        read: if  <T as Operations>::HAS_READ {
            Some(Self::read_callback)
        } else {
            <T as DefaultOperationSelection>::READ
        },

}
