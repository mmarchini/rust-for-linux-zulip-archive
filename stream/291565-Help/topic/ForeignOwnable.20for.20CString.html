        // Returns a c_char *
    }

    fn into_foreign(self) -> *mut Self::PointedTo {
        let (ptr, _, _) = self.buf.into_raw_parts();

        ptr.cast()
    }

    unsafe fn from_foreign(ptr: *mut Self::PointedTo) -> Self {
