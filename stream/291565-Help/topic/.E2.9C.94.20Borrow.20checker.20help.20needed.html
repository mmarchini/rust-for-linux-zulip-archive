struct Collection<'a> {
    field: Option<u64>,
    _p: core::marker::PhantomData<&'a ()>,
}

impl<'a> Collection<'a> {
    fn get_thing_mut(&mut self) -> Option<&mut u64> {
        self.field.as_mut()
    }

    fn set_thing(&mut self, thing: u64) {
        self.field = Some(thing);
    }
}

fn get_or_insert_new<'a>(collection: &'a mut Collection<'_>) -> &'a mut u64 {
    if collection.get_thing_mut().is_none() {
        collection.set_thing(42);
    }
    collection.get_thing_mut().unwrap()
}

fn get_or_insert_new2<'a>(collection: &'a mut Collection<'_>) -> &'a mut u64 {
    if let Some(thing) = collection.get_thing_mut() {
        thing
    } else {
        collection.set_thing(42);
        collection.get_thing_mut().unwrap()
    }
}

Is there any (safe) way I can get by with just 1 lookup when the item is present in the collection?
