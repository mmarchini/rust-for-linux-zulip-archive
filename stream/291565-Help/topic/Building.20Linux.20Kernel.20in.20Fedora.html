Install Rust and Dependencies

sudo dnf install rust rust-src bindgen-cli rustfmt clippy

Clone the Kernel

git clone git@github.com:Rust-for-Linux/linux.git

make LLVM=1 rustavailable

Install flex (and other Kconfig build deps)

sudo dnf install flex bison ncurses-devel

Install the full LLVM toolchain

sudo dnf install clang lld llvm llvm-devel

Install elfutils development package

sudo dnf install elfutils-libelf-devel

Install OpenSSL development package

sudo dnf install openssl-devel

Install QEMU

sudo dnf install qemu-system-x86

Install busybox

sudo dnf install busybox

Setup Kernel Configuration

make menuconfig

a) Go to General setup → Rust support (CONFIG_RUST), enable it. 

General setup →
    [*] Rust support (CONFIG_RUST)

b) Then under Kernel hacking → Sample kernel code → Rust samples, enable some sample modules

make rust-analyzer

It should output:

rust-project.json

Build the kernel full

make LLVM=1 -j$(nproc)

It should output:

Kernel: arch/x86/boot/bzImage is ready  (#1)

Create a clean minimal rootfs with BusyBox

#!/usr/bin/env bash
set -e
# 1. Remove old rootfs and initramfs
rm -rf minimal_initramfs initramfs.cpio initramfs.cpio.gz
# 2. Create minimal directory structure
mkdir -p minimal_initramfs/{bin,sbin,etc,proc,sys,usr/bin,usr/sbin,dev}
# 3. Copy BusyBox
cp /usr/bin/busybox minimal_initramfs/bin/
chmod +x minimal_initramfs/bin/busybox
# 4. Create device nodes (required for console)
sudo mknod -m 622 minimal_initramfs/dev/console c 5 1
sudo mknod -m 666 minimal_initramfs/dev/null c 1 3
# 5. Create BusyBox symlinks INSIDE /bin
cd minimal_initramfs/bin
for cmd in sh mount ls echo dmesg cat; do
    ln -sf busybox "$cmd"
done
cd -
# 6. Create init script
cat > minimal_initramfs/init <<'EOF'
#!/bin/sh
mount -t proc none /proc
mount -t sysfs none /sys
echo "*** Welcome to minimal initramfs! ***"
exec /bin/sh
EOF
chmod +x minimal_initramfs/init

Build the initramfs

cd minimal_initramfs
find . -print0 | cpio --null -ov --format=newc > ../initramfs.cpio
cd ..

Boot custom Linux kernel in QEMU

qemu-system-x86_64 \
  -m 2G \
  -kernel arch/x86/boot/bzImage \
  -initrd initramfs.cpio \
  -append "console=ttyS0 rdinit=/init nokaslr initcall_debug loglevel=8 earlyprintk=ttyS0" \
  -serial mon:stdio \
  -enable-kvm
