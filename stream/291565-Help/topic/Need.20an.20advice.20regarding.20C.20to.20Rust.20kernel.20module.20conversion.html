ovpn_dco_rust.rs:

#![no_std]
#![allow(improper_ctypes)]
#![allow(warnings)]

use kernel::prelude::*;
use kernel::error::{Error, code::EINVAL as EinvalKern};

include!("bindings.rs");

module! {
    type: OvpnDcoRust,
    name: "ovpn_dco_rust",
    author: "Vpnable Limited",
    description: "OpenVPN data channel offload in Rust",
    license: "GPL",
}

struct OvpnDcoRust;

impl kernel::Module for OvpnDcoRust {
    fn init(_module: &'static ThisModule) -> Result<Self> {
        pr_info!("ovpn_dco_rust module loaded.\n");

        let err = unsafe { ovpn_tcp_init() };
        if err != 0 {
            pr_err!("ovpn_dco_rust: Failed to initialize TCP subsystem\n");
            return Err(EinvalKern);
        }

        let err = unsafe { ovpn_netlink_register() };
        if err != 0 {
            pr_err!("ovpn_dco_rust: Failed to register netlink\n");
            return Err(EinvalKern);
        }

        Ok(OvpnDcoRust)
    }
}

impl Drop for OvpnDcoRust {
    fn drop(&mut self) {
        pr_info!("ovpn_dco_rust module unloaded.\n");

        unsafe {
            ovpn_netlink_unregister();
        }
    }
}

#[no_mangle]
extern "C" {
    fn ovpn_tcp_init() -> i32;
    fn ovpn_netlink_register() -> i32;
    fn ovpn_netlink_unregister();
}
