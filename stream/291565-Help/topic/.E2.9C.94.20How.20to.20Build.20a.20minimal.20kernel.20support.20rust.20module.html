Regarding this problem, I have solved it:

error: target file "./scripts/target.json" does not exist

When compiling the arm64 module, need to add ARCH=arm64 to the Makefile.
like this:

KDIR ?= ~/Linux/rust_for_linux/next_0423/linux

obj-m := hello_world.o

all:
    $(MAKE) -C $(KDIR) M=$(PWD) LLVM=1 ARCH=arm64 modules

clean:
    $(MAKE) -C $(KDIR) M=$(PWD) clean
