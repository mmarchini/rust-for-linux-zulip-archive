impl_has_work! {
    impl{T} HasWork<Self> for SubmitWorker<T> { self.work }
}

impl<T> WorkItem for SubmitWorker<T> {
    type Pointer = Arc<SubmitWorker<T>>;

    fn run(this: Arc<SubmitWorker<T>>) {
        let guard = this.jobq.try_access();
        if guard.is_none() {
            // If we're here, jobq is tearing down. Further submissions are ignored.
            return;
        }
        let jobq = guard.unwrap();

        let mut jobq = jobq.lock();

        jobq.capacity = run_all_ready_jobs(&mut jobq, this.run_job);
    }
}
