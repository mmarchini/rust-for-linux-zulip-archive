Hello, so injecting global flags directly into Rust toolchain for example clippy-driver is ​​not currently supported natively right? So overriding it with writeShellScriptBin is a common practice. for example:

let
tools = [
{ name = "clippy-driver"; cmd = "clippy-driver"; flags = "-Dwarnings"; }
{ name = "rustdoc"; cmd = "rustdoc"; flags = "-Dwarnings"; }
];
in
builtins.listToAttrs (map (tool:
{
name = tool.name;
value = pkgs.writeShellScriptBin tool.name ''
exec ${rustToolchain}/bin/${tool.cmd} ${tool.flags} "$@"
'';
}) tools)
