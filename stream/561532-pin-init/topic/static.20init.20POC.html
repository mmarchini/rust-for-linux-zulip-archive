Here's a POC for static init using const_trait_impl:

#![feature(const_trait_impl)]

use std::convert::Infallible;
use std::mem::MaybeUninit;

pub const unsafe trait StaticInit<T: ?Sized>: Sized {
    // Initialize a `T` and place it into `slot`, with the assumption
    // that it will be copied to `addr`.
    // Takes `slot` instead of returning `T` so a run-time init is just
    // `expr.__init(slot, slot)`.
    unsafe fn __init(self, addr: *mut T, slot: *mut T);
}

struct List {
    prev: *mut List,
    next: *mut List,
}

unsafe impl Sync for List {}

struct ListInit;

const unsafe impl StaticInit<List> for ListInit {
    unsafe fn __init(self, addr: *mut List, slot: *mut List) {
        unsafe {
            (*slot).next = addr;
            (*slot).prev = addr;
        }
    }
}

macro_rules! static_pin_init {
    (static $name:ident: $ty:ty = $expr:expr) => {
        static $name: $ty = {
            let init = $expr;
            unsafe {
                let mut slot = MaybeUninit::uninit();
                StaticInit::__init(init, (&raw const $name).cast_mut(), slot.as_mut_ptr());
                slot.assume_init()
            }
        };
    }
}

static_pin_init!(static LIST: List = ListInit);
