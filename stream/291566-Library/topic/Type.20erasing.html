Gary Guo (Mar 27 2026 at 14:22):

It occurs to me that type erasing is something that DRM fences would need. I've done some POC on getting a generic API:

#![feature(unsize)]
#![feature(ptr_metadata)]

use core::ptr::Pointee;
use core::ops::Deref;
use core::marker::Unsize;
use core::ops::DerefMut;
use core::marker::PhantomData;

pub trait Pointer {
    type Target: ?Sized;

    fn into_raw(this: Self) -> *mut Self::Target;
    unsafe fn from_raw(ptr: *mut Self::Target) -> Self;
}

impl<T: ?Sized> Pointer for Box<T> {
    type Target = T;

    #[inline]
    fn into_raw(this: Self) -> *mut T {
        Box::into_raw(this)
    }

    #[inline]
    unsafe fn from_raw(ptr: *mut T) -> Self {
        unsafe { Box::from_raw(ptr) }
    }
}

trait Erasable {
    /// Recover metadata from a raw pointer.
    ///
    /// # Safety
    ///
    /// Exists `T: Unsize<Self>`, where `ptr` is pointer to a valid `T`.
    unsafe fn unerase(ptr: *mut ()) -> *mut Self;
}

#[repr(C)]
struct ErasableWrapper<T: ?Sized, U: ?Sized> {
    meta: <U as Pointee>::Metadata,
    data: T,
}

type ErasedWrapper<U> = ErasableWrapper<U, U>;

impl<U: ?Sized> Erasable for ErasedWrapper<U> {
    #[inline]
    unsafe fn unerase(ptr: *mut ()) -> *mut Self {
        // SAFETY: for all `T`, `ErasableWrapper<T, U>` start with the same `meta`.
        let meta = unsafe { (ptr as *const <U as Pointee>::Metadata).read() };
        core::ptr::from_raw_parts_mut(ptr, meta)
    }
}

impl<T, U: ?Sized> ErasableWrapper<T, U> {
    #[inline]
    pub fn new(data: T) -> Self where T: Unsize<U> {
        let meta = core::ptr::metadata(&data as &U);
        Self {
            meta, data
        }
    }
}

impl<T: ?Sized, U: ?Sized> Deref for ErasableWrapper<T, U> {
    type Target = T;

    #[inline]
    fn deref(&self) -> &T {
        &self.data
    }
}

impl<T: ?Sized, U: ?Sized> DerefMut for ErasableWrapper<T, U> {
    #[inline]
    fn deref_mut(&mut self) -> &mut T {
        &mut self.data
    }
}

struct Thin<P: Pointer> where P::Target: Erasable {
    ptr: *mut (),
    p: PhantomData<P>,
}

impl<T: Erasable + ?Sized, P: Pointer<Target = T>> Drop for Thin<P> {
    #[inline]
    fn drop(&mut self) {
        let ptr = unsafe { T::unerase(self.ptr) };
        drop(unsafe { P::from_raw(ptr) });
    }
}

impl<T: Erasable + ?Sized, P: Pointer<Target = T>> Thin<P> {
    #[inline]
    fn new(p: P) -> Self {
        Self {ptr: Pointer::into_raw(p).cast() , p: PhantomData }
    }
}

impl<T: Erasable + ?Sized, P: Pointer<Target = T>> Deref for Thin<P> {
    type Target = T;

    #[inline]
    fn deref(&self) -> &T {
        unsafe { &*T::unerase(self.ptr) }
    }
}

impl<T: Erasable + ?Sized, P: Pointer<Target = T>> DerefMut for Thin<P> {
    #[inline]
    fn deref_mut(&mut self) -> &mut T {
        unsafe { &mut *T::unerase(self.ptr) }
    }
}

trait Fence {
    fn act_on_data(&self);
}

impl Fence for u32 {
    fn act_on_data(&self) {
        println!("{}", self);
    }
}

impl Fence for () {
    fn act_on_data(&self) {
        println!("unit");
    }
}

fn make_thin_1() -> Thin<Box<ErasedWrapper<dyn Fence>>> {
    Thin::new(Box::new(ErasableWrapper::new(1u32)))
}

fn make_thin_2() -> Thin<Box<ErasedWrapper<dyn Fence>>> {
    Thin::new(Box::new(ErasableWrapper::new(())))
}

fn use_thin(t: Thin<Box<ErasedWrapper<dyn Fence>>>) {
    t.act_on_data()
}

fn main() {
    assert_eq!(size_of::<Thin<Box<ErasedWrapper<dyn Fence>>>>(), size_of::<usize>());
    use_thin(make_thin_1());
    use_thin(make_thin_2());
}
