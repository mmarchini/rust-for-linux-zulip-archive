/// Box extension Proiding functions to pass GFP flags to the allocator.
pub trait BoxAllocFlagExt<T: Sized, A: FlagAllocator> {
    /// Allocate uninit box with flags.
    fn try_new_uninit_flag_in(flags: bindings::gfp_t, alloc: A) -> Result<Box<core::mem::MaybeUninit<T>, A>, AllocError>;

    /// Allocated uninit box with flags on Global Allocator.
    fn try_new_uninit_flag(flags: bindings::gfp_t) -> Result<Box<core::mem::MaybeUninit<T>, A>, AllocError>
        where
        A: alloc::alloc::Global
    {
        Self::try_new_uninit_flag_in(flags, alloc::alloc::Global)
    }
}

This still change the type of the box and the box would have to be freed with the FlagAllocator. But it would be fine to free with the default allocator.

pub trait BoxExt<T: ?Sized> {
    fn try_new_atomic(x: T) -> Result<Self, AllocError>
    where
        Self: Sized;
}

impl<T> BoxExt<T> for Box<T> {
    fn try_new_atomic(x: T) -> Result<Box<T>, AllocError> {
        let layout = core::alloc::Layout::new::<core::mem::MaybeUninit<T>>();
        let ptr = crate::ALLOCATOR_ATOMIC.allocate(layout)?.cast();
        let mut boxed: Box<MaybeUninit<T>> =
            unsafe { Box::from_raw_in(ptr.as_ptr(), alloc::alloc::Global) };

        unsafe {
            boxed.as_mut_ptr().write(x);
            Ok(boxed.assume_init())
        }
    }
}

With this approach we could use whatever allocation function we want and still free with the global allocator without changing the type of the box. This is only atomic, but we could add a param for the flag.
