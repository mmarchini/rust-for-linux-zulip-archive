And RcuFreeSafe type being:

/// # Safety
/// Implementing this trait means it's safe to call `drop_in_place_before_gp()` on
/// the object and wait for one grace period to free it.
pub unsafe trait RcuFreeSafe {
    unsafe fn drop_in_place_before_gp(*mut Self);
}

and 

unsafe impl RcuFreeSafe for PollCondVar {
    unsafe fn drop_in_place_before_gp(ptr: *mut Self) {
        unsafe { bindings::__wake_up_pollfree((*ptr).inner.wait_queue_head.get()) };
    }
}

you can reuse the above function in PollCondVar::drop() as well.
