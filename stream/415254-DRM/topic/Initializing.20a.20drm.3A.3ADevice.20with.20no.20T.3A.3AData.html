It's a bit unfortunate that drm::Device::new() takes T::Data, since T::Data may need the Device to initialize itself.

Consider this excerpt from the Asahi project:

        let drm = drm::device::Device::<AsahiDriver>::new_no_data(pdev.as_ref())?;
        let gpu = match (cfg.gpu_gen, cfg.gpu_variant, compat.as_slice()) {
            (hw::GpuGen::G13, _, &[12, 3, 0]) => {
                gpu::GpuManagerG13V12_3::new(&drm, &res, cfg)? as Arc<dyn gpu::GpuManager>     <----------

       <...snip>

            _ => {
                dev_info!(
                    pdev.as_ref(),
                    "Unsupported GPU/firmware combination ({:?}, {:?}, {:?})\n",
                    cfg.gpu_gen,
                    cfg.gpu_variant,
                    compat
                );
                return Err(ENODEV);
            }
        };

        let data = Arc::pin_init(
            try_pin_init!(AsahiData {
                gpu,
                pdev: pdev.clone(),
                resources: res,
            }),
            GFP_KERNEL,
        )?;

They added a hack to make this possible, i.e.: drm::Device::new_no_data() and drm::Device::init_data(). I think we should follow suit and somehow allow this upstream.

How about this?

    pub fn new_from_closure<F>(dev: &device::Device, data: F) -> Result<ARef<Self>>
    where
        F: FnOnce(ARef<Self>) -> T::Data,
    {
        let drm = Self::create_drm_device(dev)?;
        let data = data(drm.clone());
        drm.set_data(data);
        Ok(drm)
    }
