/// A GEM object whose private-data layout is not known.
///
/// Not all GEM objects are created equal, and subsequently drivers may occasionally need to deal
/// with situations where they are working with a GEM object whose [`DriverObject`] implementation
/// is ambiguous, or simply doesn't exist (e.g. a GEM object created from an ffi call).
///
/// It may be used just like a normal [`Object`], with the exception that it cannot access
/// driver-private data.
///
/// # Invariant
///
/// Via `#[repr(transparent)]`, this type is guaranteed to have an identical data layout to
/// `struct drm_gem_object`.
#[repr(transparent)]
pub struct OpaqueObject<T: drm::Driver>(Opaque<bindings::drm_gem_object>, PhantomData<T>);

impl<T: drm::Driver> IntoGEMObject for OpaqueObject<T> {
    type Driver = T;

    unsafe fn as_ref<'a>(self_ptr: *mut bindings::drm_gem_object) -> &'a Self {
        // SAFETY:
        // - This cast is safe via our type invariant.
        // - `self_ptr` is guaranteed to be a valid pointer to a gem object by our safety contract.
        unsafe { &*self_ptr.cast().cast_const() }
    }

    fn as_gem_obj(&self) -> *mut bindings::drm_gem_object {
        self.0.get()
    }
}

impl<D, T> AsRef<OpaqueObject<D>> for T
where
    T: IntoGEMObject<Driver = D>,
    D: drm::Driver,
{
    fn as_ref(&self) -> &OpaqueObject {
        // SAFETY: This cast is safe via our type invariant.
        unsafe { &*self.as_gem_obj().cast_const() }
    }
}

///
/// # Examples
///
/// Where `map` is an `IoSysMapRef` which contains 3 `Foo`s.
///
/// ```
/// # use kernel::iosys_map::{*, tests::*};
/// ##[derive(Default)]
/// struct Foo {
///     a: u8,
/// };
///
/// // SAFETY: All bit patterns are acceptable values for `Foo`.
/// unsafe impl kernel::transmute::FromBytes for Foo{};
/// // SAFETY: Instances of `Foo` have no uninitialized portions.
/// unsafe impl kernel::transmute::AsBytes for Foo{};
///
/// # let map = VecIoSysMap::<Foo>::new(3).unwrap();
/// let whole = kernel::iosys_map_read!(map[2]);
/// let field = kernel::iosys_map_read!(map[1].a);
/// ```
#[macro_export]
macro_rules! iosys_map_read {
    ($map:expr, $idx:expr, $($field:tt)*) => {{
        (|| -> ::core::result::Result<_, $crate::error::Error> {
            let base_offset = $crate::iosys_map::IoSysMapRef::item_from_index(&$map, $idx)?;

            // SAFETY: `base_offset` came from item_from_index(), which always returns a valid
            // offset within the iosys map.
            let base_ptr = unsafe {
                $crate::iosys_map::IoSysMapRef::ptr_from_offset(&$map, base_offset)
            };

            // SAFETY: `ptr_from_index()` always returns a valid pointer to an item within the
            // memory region pointed to by the iosys mapping.
            let field_ptr = unsafe { &raw const (*base_ptr) $($field)* };

            // SAFETY: field_ptr points to a field contained within the struct pointed to by
            // item_ptr, so we know that field_ptr is always located after item_ptr and thus this
            // subtraction will not underflow.
            let field_offset = unsafe { field_ptr.byte_offset_from(base_ptr) } as usize;

            // SAFETY:
            // - Both field_ptr and item_ptr are valid pointers in the iosys map, and contained
            //   within the same allocation.
            // - `base_offset` is derived from item_from_index(), which always returns a valid
            //   offset within the iosys map.
            // - Since we derived `field_offset` from `field_ptr - item_ptr`, we know `field_offset`
            //   does not go out of the bounds of a single item within the iosys map.
            // - Thus, `base_offset + field_offset` are guaranteed to refer to a valid offset within
            //   the iosys_map that corresponds to the field we are trying to write to.
            // - Additionally, since the resulting offset will be valid we are guaranteed that this
            //   addition will not overflow.
            unsafe {
                let offset = base_offset.unchecked_add(field_offset);

                // (confirm that we're correct and field_offset truly does not go out of bounds)
                debug_assert!(offset < $map.size());

                ::core::result::Result::Ok(
                    $crate::iosys_map::IoSysMapRef::field_read(&$map, offset)
                )
            }
        })()
    }};
    ($map:ident [ $idx:expr ] $($field:tt)*) => {
        $crate::iosys_map_read!($map, $idx, $($field)*)
    };
    ($($map:ident).* [ $idx:expr ] $($field:tt)*) => {
        $crate::iosys_map_read!($($map).*, $idx, $($field)*)
    };
}
