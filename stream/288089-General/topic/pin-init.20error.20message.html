The problem in this code was a missing #[pin]. I had:

#[pin_data]
struct SampleModule {
    _fs: fs::Registration,
}

impl kernel::InPlaceModule for SampleModule {
    fn init(module: &'static ThisModule) -> impl PinInit<Self, Error> {
        try_pin_init!(Self {
            _fs <- fs::Registration::new::<SampleFs>(module),
        })
    }
}

And the fix was simply to add #[pin] to _fs:

#[pin_data]
struct SampleModule {
    #[pin]
    _fs: fs::Registration,
}
