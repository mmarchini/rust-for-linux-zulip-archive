impl<const B: usize> Bounded<B> {
    fn shr<const SHIFT: usize, const RES: usize>(self) -> Bounded<RES> {
        const { assert!(B == RES + SHIFT) }
        Bounded(self.0 >> SHIFT)
    }
    fn shl<const SHIFT: usize, const RES: usize>(self) -> Bounded<RES> {
        const { assert!(B + SHIFT == RES) }
        Bounded(self.0 << SHIFT)
    }
}
