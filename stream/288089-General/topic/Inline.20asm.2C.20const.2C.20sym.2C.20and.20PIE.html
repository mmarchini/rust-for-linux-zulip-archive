    let mut addr: *const i32;

    unsafe {
        asm!(
            "lea {0}, [{1}]",

            sym MY_GLOBAL,
        );
    }
    println!("{:?}", addr);

fails with

relocation R_X86_64_32S against .rodata._ZN10playground9MY_GLOBAL17h21cfec9b2d52d43cE can not be used when making a PIE object; recompile with -fPIE

And then it can be fixed by doing this instead:

use std::arch::asm;

static MY_GLOBAL: i32 = 10;

fn main() {
    let mut addr: *const i32;
    unsafe {
        asm!(
            "lea {1}(%rip), {0}",
            out(reg) addr,
            sym MY_GLOBAL,
            options(att_syntax)
        );
    }
    assert_eq!(addr, &MY_GLOBAL as *const i32);
}

Is this the same issue? Perhaps the fact that it is a problem for sym too could mean something interesting for the const ptr case too? Thoughts?
