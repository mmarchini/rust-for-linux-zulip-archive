I've been thinking more about custom formatting, and I wonder if this approach could work:

use core::fmt;

/// Formatter that you can write invalid utf-8 to.
struct KernelFormatter<'a> {
    inner: fmt::Formatter<'a>,
}

impl<'a> KernelFormatter<'a> {
    unsafe fn new<'b>(f: &'b mut fmt::Formatter<'a>) -> &'b mut Self {
        unsafe { &mut *(f as *mut _ as *mut Self) }
    }

    fn write_bytes(&mut self, byt: &[u8]) -> fmt::Result {
        self.inner.write_str(unsafe { core::str::from_utf8_unchecked(byt) })
    }
}

/// Does not assume that `write_str` is passed values containing valid utf-8.
unsafe trait KernelWrite: core::fmt::Write {
    fn write_bytes(&mut self, byt: &[u8]) -> fmt::Result {
        self.write_str(unsafe { core::str::from_utf8_unchecked(byt) })
    }
}

trait KernelDisplay {
    fn fmt(&self, f: &mut KernelFormatter<'_>) -> fmt::Result;
}



struct FmtAdapter<T>(T);

impl<T: KernelDisplay> core::fmt::Display for FmtAdapter<T> {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        self.0.fmt(unsafe { KernelFormatter::new(f) })
    }
}

fn main() {
    // Our printing macros just wrap every argument in FmtAdapter
    // and otherwise pass through to built-in formatting logic.
    println!("{}", FmtAdapter(17));
}

Then, we implement our own KernelDisplay trait for the types we wish to be able to print. This lets us print core::str::CStr directly, while also letting us remove the ability to print raw pointers entirely. Thoughts?

pub struct FmtAdapter<T>(pub T);

macro_rules! impl_fmt_adapter {
    ($($trait:ident),* $(,)?) => {
        $(
            impl<T: fmt::$trait> fmt::$trait for $crate::str::FmtAdapter<T> {
                fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
                    let Self(t) = self;
                    fmt::$trait::fmt(t, f)
                }
            }
        )*
    };
}

impl_fmt_adapter!(Debug, LowerHex, UpperHex, Octal, Binary, Pointer, LowerExp, UpperExp);

trait KernelDisplay {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result;
}

impl<T: fmt::Display> KernelDisplay for T {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        fmt::Display::fmt(self, f)
    }
}

impl<T: KernelDisplay> fmt::Display for FmtAdapter<T> {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        let Self(t) = self;
        KernelDisplay::fmt(t, f)
    }
}

pub trait KernelDisplay {
    /// Formats the value using the given formatter, like [`fmt::Display::fmt`].
    ///
    /// This method is invoked by the [`fmt!`] macro through the [`FmtAdapter`] wrapper.
    ///
    /// [`fmt!`]: crate::prelude::fmt!
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result;
}

impl<T: fmt::Display> KernelDisplay for T {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        fmt::Display::fmt(self, f)
    }
}

/// Internal adapter used to route formatting through [`KernelDisplay`] inside the [`fmt!`] macro.
///
/// It is inserted automatically by the [`fmt!`] macro and is not meant to be used directly.
///
/// [`fmt!`]: crate::prelude::fmt!
#[doc(hidden)]
pub struct FmtAdapter<T>(pub T);

macro_rules! impl_fmt_adapter {
    ($($trait:ident),* $(,)?) => {
        $(
            impl<T: fmt::$trait> fmt::$trait for $crate::str::FmtAdapter<T> {
                fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
                    let Self(t) = self;
                    fmt::$trait::fmt(t, f)
                }
            }
        )*
    };
}

impl_fmt_adapter!(Debug, LowerHex, UpperHex, Octal, Binary, Pointer, LowerExp, UpperExp);

impl<T: KernelDisplay> fmt::Display for FmtAdapter<&T> {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        let Self(t) = self;

    }
}
