Alice Ryhl (May 29 2024 at 18:42):

Can I have an assert macro that takes a struct and a field name and asserts that the field is structurally pinned with #[pin]?

Benno Lossin (May 29 2024 at 19:04):

I have a very hacky solution:

macro_rules! assert_pinned {
    ($ty:ty, $field:ident, $field_ty:ty) => {
        fn _assert_pinned<T, E, I>(_: impl FnOnce(*mut T, I) -> Result<(), E>)
        where
            I: ::pinned_init::PinInit<T, E>,
        {
        }
        fn _check<I>()
        where
            I: ::pinned_init::PinInit<$field_ty, ()>,
        {
            let data = unsafe { <$ty as ::pinned_init::__internal::HasPinData>::__pin_data() };
            _assert_pinned::<$field_ty, (), I>(|ptr, init| unsafe { data.$field(ptr, init) });
        }
    };
}
