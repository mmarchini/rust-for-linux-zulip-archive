Topic: pin-init references to initialized fields

Benno Lossin (May 30 2025 at 09:28):

@Danilo Krummrich has a use-case for references to already initialized fields within the initializer macros. @Alice Ryhl also said this to be useful at RustWeek. I also think it's useful, but I fear that this is a footgun, since any raw pointer derived from such a reference-to-a-field might get invalidated upon using a reference to the entire struct.

Example

use std::{marker::PhantomPinned, pin::Pin};

use pin_init::{pin_data, pin_init, InPlaceInit};

#[pin_data]
#[derive(Debug)]
struct SelfWriter {
    val: u8,
    ptr: *mut u8,
    #[pin]
    _pin: PhantomPinned,
}

impl SelfWriter {
    fn write(self: Pin<&mut Self>, val: u8) {
        unsafe { self.ptr.write(val) };
    }

    fn write2(self: Pin<&mut Self>, val: u8) {
        let this = unsafe { Pin::into_inner_unchecked(self) };
        this.val = val;
    }
}

fn main() {
    let x = pin_init!(&this in SelfWriter {
        val: 0,
        ptr: {
            // this would be generated by the macro, so one could just write `ptr: val` here
            // instead.
            let val = unsafe { &mut (*this.as_ptr()).val };
            val
        },
        _pin: PhantomPinned,
    });
    let mut x = Box::pin_init(x).unwrap();
    println!("{x:?}");
    x.as_mut().write(42);
    println!("{x:?}");
    x.as_mut().write2(82);
    println!("{x:?}");
    x.as_mut().write(42);
    println!("{x:?}");
}

If I use UnsafePinned in the example, then it does continue to work:

Example with UnsafePinned

#![feature(unsafe_pinned)]

use std::pin::{Pin, UnsafePinned};

use pin_init::{pin_data, pin_init, InPlaceInit};

#[pin_data]
#[derive(Debug)]
struct SelfWriter {
    #[pin]
    val: UnsafePinned<u8>,
    ptr: *mut u8,
}

impl SelfWriter {
    fn write(self: Pin<&mut Self>, val: u8) {
        unsafe { self.ptr.write(val) };
    }

    fn write2(self: Pin<&mut Self>, new: u8) {
        let val = unsafe { self.map_unchecked_mut(|t| &mut t.val) };
        unsafe { val.get_mut_pinned().write(new) };
    }
}

fn main() {
    let x = pin_init!(&this in SelfWriter {
        val: UnsafePinned::new(0),
        ptr: {
            // this would be generated by the macro, so one could just write `ptr:
            // val.get_mut_pinned()` here instead.
            let val: = unsafe { Pin::new_unchecked(&mut (*this.as_ptr()).val) };
            val.get_mut_pinned()
        },
    });
    let mut x = Box::pin_init(x).unwrap();
    println!("{x:?}");
    x.as_mut().write(42);
    println!("{x:?}");
    x.as_mut().write2(82);
    println!("{x:?}");
    x.as_mut().write(42);
    println!("{x:?}");
}
