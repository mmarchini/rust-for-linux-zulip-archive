Gary Guo (Apr 13 2026 at 15:10):

The general issue for SysmemFlush (https://lore.kernel.org/rust-for-linux/20260409-fix-systemflush-v1-1-a1d6c968f17c@nvidia.com/) is that one resource depends on another resource and it needs to outlive its dependencies so Drop can still access it.

I've done some experiments and I think something like this could work:

use std::marker::PhantomData;

pub trait ForLt {
    type Ty<'l>;
}

#[doc(hidden)]
pub struct ForLtImpl<F>(F);

mod internal {

use super::*;
    pub trait WithLt<'l> {
        type Ty;
    }

    impl<'lt, F, O> WithLt<'lt> for ForLtImpl<F>
    where
        F : FnOnce(&'lt ()) -> O
    {
        type Ty = O;
    }

    impl<T: for<'l> WithLt<'l>> ForLt for T {
        type Ty<'l> = <T as WithLt<'l>>::Ty;
    }
}

macro_rules! for_lt {
    (for <$lt:lifetime> $t: ty) => {
        ForLtImpl::<for<$lt> fn(&$lt ()) -> $t>
    };
    ($t: ty) => {
        ForLtImpl::<fn(&'_ ()) -> $t>
    }
}

// All of the above to get higher-ranked lifetimes for types, not just trait bounds.

struct Bar;

struct SysmemFlush<'a>(
    &'a Bar,
);

impl<'a> Drop for SysmemFlush<'a> {
    fn drop(&mut self) {
        // Still have access to `bar`.
    }
}

struct DevResLt<T: ForLt> {
    // Actually `T::Ty<'a>` for some `'a`
    data: PhantomData<T::Ty<'static>>,
}

// Having driver core making sure `SysmemFlush` is dropped before dropping `bar`.

type Foo = DevResLt<for_lt!(for <'a> SysmemFlush<'a>)>;

If only drop is needed you could probably use with what I have above with something like:

struct DevResInner<'a, T,  D> {
    dep: &'a D,
    data: T,
}

impl<T: Send + DevresDrop<D>, D: Send> Drop for DevResInner<T, D> {
    fn drop(&mut self) {
        self.data.devres_drop(self, self.dep);
    }
}

type DevRes<T, D = ()> = DevResLt<for_lt!(for<'a> DevResInner<'a T, D>)>;
