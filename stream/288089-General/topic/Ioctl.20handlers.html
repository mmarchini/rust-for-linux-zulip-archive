So, there is discussion going on related to how we can make nice ioctl handlers on the list.

I had the idea of trying to mirror how axum does routing when it receives incoming http requests. It's able to do it without any macros.

Sketch:

// SPDX-License-Identifier: GPL-2.0

// Copyright (C) 2024 Google LLC.

//! Ioctl dispatcher.

use kernel::ioctl::_IOC_NR;
use kernel::uapi::{_IOC_READ, _IOC_WRITE};

pub struct IoctlDispatcher<T> {
    me: T,
    cmd: u32,
    cmd_nr: u32,
    arg: usize,
    ret: isize,
    skip: bool,
}

impl<T> IoctlDispatcher<T> {
    #[inline]
    pub fn new(me: T, cmd: u32, arg: usize) -> Self {
        Self {
            me,
            cmd,
            cmd_nr: _IOC_NR(cmd),
            arg,
            // This value is used if no ioctl matches.
            ret: EINVAL.to_errno(),
            skip: false,
        }
    }

    #[inline]
    #[must_use = "You must call .finish() after your last ioctl handler."]
    pub fn iowr(&mut self, nr: u32, handler: impl IoctlWR) -> &mut Self {
        // Fail the build if `nr` is not compatible with the provided handler.
        build_assert!(_IOC_READ|_IOC_WRITE == _IOC_DIR(nr));

        if self.skip || _IOC_NR(nr) != self.cmd_nr {
            return self;
        }

        self.skip = true;
        self.ret = handler.handle(self.me, self.cmd, self.arg);
        self
    }

    #[inline]
    pub fn finish(&mut self) -> isize {
        self.ret
    }
}

To dispatch, you use it like this:

IoctlDispatcher::new(self, cmd, arg)
    .iow(bindings::BINDER_SET_MAX_THREADS, Self::set_max_threads),
    .iow(bindings::BINDER_THREAD_EXIT, Self::on_thread_exit),
    .iowr(bindings::BINDER_GET_NODE_DEBUG_INFO, Self::get_node_debug_info)
    .iowr(bindings::BINDER_GET_FROZEN_INFO, Self::get_frozen_info)
    .finish()

We then have blanket implementations for the IoctlWr trait similar to the axum Handler trait. It lets us write function signatures like this:

struct Write<T>(pub T);
struct WriteRead<T>(pub T);
struct Inout<'a, T>(pub &'a mut T);

impl MyType {
    fn set_max_threads(&self, Write(threads): Write(u32)) { ... }
    fn on_thread_exit(&self) -> Result { ... }
    fn get_node_debug_info(&self, WriteRead(info): WriteRead<BinderNodeInfoForRef>) -> Result<BinderNodeInfoForRef> { ... }
    fn get_frozen_info(&self, Inout(info): Inout<'_, BinderFrozenStatusInfo>) -> Result { ... }
}
