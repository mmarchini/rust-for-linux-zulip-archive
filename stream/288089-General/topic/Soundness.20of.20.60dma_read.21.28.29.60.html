struct Foo {

struct OffsetIn<T> {
    offset: usize,
    _in: PhantomData<T>,
}
impl<T> OffsetIn<T> {
    fn create() -> Self {
        Self {
            offset: 0,
            _in: PhantomData,
        }
    }

    fn get_uninit(&self) -> MaybeUninit<T> {
        MaybeUninit::uninit()
    }

    fn get_offset_for(self, _t: &T) -> usize {
        self.offset
    }
}

}
