Gary Guo (Feb 24 2026 at 07:12):

This looks like an issue of our Makefile where the depinfo is moved into place rather than using --emit=depinfo=$(depinfo).

@Onur Özkan can you see if this fix the issue for you?

diff --git a/rust/Makefile b/rust/Makefile
index 629b3bdd2b20..e603b194366b 100644
--- a/rust/Makefile
+++ b/rust/Makefile
@@ -526,11 +526,10 @@ quiet_cmd_rustc_procmacrolibrary = $(RUSTC_OR_CLIPPY_QUIET) PL $@
       cmd_rustc_procmacrolibrary = \
    $(if $(skip_clippy),$(RUSTC),$(RUSTC_OR_CLIPPY)) \
        $(filter-out $(skip_flags),$(rust_common_flags) $(rustc_target_flags)) \
-       --emit=dep-info,link --crate-type rlib -O \
+       --emit=dep-info=$(depfile) --emit=link=$@ \
+       --crate-type rlib -O \
        --out-dir $(objtree)/$(obj) -L$(objtree)/$(obj) \
-       --crate-name $(patsubst lib%.rlib,%,$(notdir $@)) $<; \
-   mv $(objtree)/$(obj)/$(patsubst lib%.rlib,%,$(notdir $@)).d $(depfile); \
-   sed -i '/^\#/d' $(depfile)
+       --crate-name $(patsubst lib%.rlib,%,$(notdir $@)) $<

 $(obj)/libproc_macro2.rlib: private skip_clippy = 1
 $(obj)/libproc_macro2.rlib: private rustc_target_flags = $(proc_macro2-flags)

Our initial rust Makefile used the same approach, but it looks like in 295d8398c67e314d99bb070f38883f83fe94a97a Masahiro changes existing things to use explicit path emit. The comment removal is also no longer necessary after 2185242faddd12a1ba1060be5caf584fe5aba93a.
